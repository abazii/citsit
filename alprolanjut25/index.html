1. DEKLARASI STRUCT & VARIABEL GLOBAL 
const int MAX_MAHASISWA = 100;
const string FILE_DB = "data_pendaftar_magang.txt";

struct TanggalLahir {
    int tgl;
    string bulan;
    int tahun;
};

struct Mahasiswa {
    int nim;
    string nama;
    string jenisKelamin;
    TanggalLahir tglLahir;
    int semester;
    string prodi;
    string fakultas;
    float ipk;
};
2. FUNGSI UTILITAS: `clearBuffer` & `cekNIM` 
void clearBuffer() {
    cin.clear();
    cin.ignore(10000, '\n');
}

bool cekNIM(int cekNim) {
    ifstream fileInput(FILE_DB);
    if (!fileInput.is_open()) return false;

    Mahasiswa data;
    // Membaca menggunakan extraction operator >> (Standard Spasi)
    while (fileInput >> data.nim >> data.nama >> data.jenisKelamin 
                     >> data.tglLahir.tgl >> data.tglLahir.bulan >> data.tglLahir.tahun 
                     >> data.semester >> data.prodi >> data.fakultas >> data.ipk) {
        if (data.nim == cekNim) {
            fileInput.close();
            return true;
        }
    }
    fileInput.close();
    return false;
}
3. FILE WRITING: `appendKeFile` & `tulisSemuaKeFile` 
void appendKeFile(const Mahasiswa *m) {
    ofstream fileOutput(FILE_DB, ios::app);
    if (fileOutput.is_open()) {
        // Menggunakan spasi " " sebagai pemisah data
        fileOutput << m->nim << " " << m->nama << " " << m->jenisKelamin << " "
                   << m->tglLahir.tgl << " " << m->tglLahir.bulan << " " << m->tglLahir.tahun << " "
                   << m->semester << " " << m->prodi << " " << m->fakultas << " " << m->ipk << "\n";
        fileOutput.close();
        cout << ">> [SUCCESS] Data berhasil ditambahkan ke file." << endl;
    } else {
        cout << ">> Gagal membuka file!" << endl;
    }
}

// Menulis ulang seluruh data (Spasi sebagai pemisah data)
void tulisSemuaKeFile(Mahasiswa arr[], int n) {
    ofstream fileOutput(FILE_DB); // Menimpa file asli
    if (fileOutput.is_open()) {
        for (int i = 0; i < n; i++) {
            fileOutput << arr[i].nim << " " << arr[i].nama << " " << arr[i].jenisKelamin << " "
                       << arr[i].tglLahir.tgl << " " << arr[i].tglLahir.bulan << " " << arr[i].tglLahir.tahun << " "
                       << arr[i].semester << " " << arr[i].prodi << " " << arr[i].fakultas << " " << arr[i].ipk << "\n";
        }
        fileOutput.close();
    }
}

5. FILE READING & UNDERSCORE-TO-SPACE PRINT 
int loadDataToArray(Mahasiswa arr[]) {
    ifstream fileInput(FILE_DB);
    if (!fileInput.is_open()) return 0;

    int jumlah = 0;
    // Membaca file baris demi baris menggunakan separator spasi
    while (jumlah < MAX_MAHASISWA && 
           (fileInput >> arr[jumlah].nim >> arr[jumlah].nama >> arr[jumlah].jenisKelamin 
                      >> arr[jumlah].tglLahir.tgl >> arr[jumlah].tglLahir.bulan >> arr[jumlah].tglLahir.tahun 
                      >> arr[jumlah].semester >> arr[jumlah].prodi >> arr[jumlah].fakultas >> arr[jumlah].ipk)) {
        jumlah++;
    }
    fileInput.close();
    return jumlah;
}

void cetakData(Mahasiswa m) {
    // Mengembalikan '_' menjadi spasi saat menampilkan data agar rapi
    for(char &c : m.nama) if(c == '_') c = ' ';
    for(char &c : m.jenisKelamin) if(c == '_') c = ' ';
    for(char &c : m.tglLahir.bulan) if(c == '_') c = ' ';
    for(char &c : m.prodi) if(c == '_') c = ' ';
    for(char &c : m.fakultas) if(c == '_') c = ' ';

    cout << "NIM       : " << m.nim << endl;
    cout << "Nama      : " << m.nama << endl;
    cout << "Lahir     : " << m.tglLahir.tgl << " " << m.tglLahir.bulan << " " << m.tglLahir.tahun << endl;
    cout << "Prodi     : " << m.prodi << endl << "Fakultas  : " << m.fakultas << endl;
    cout << "Semester  : " << m.semester << endl << "IPK       : " << m.ipk << endl;
}
6. SEARCHING & DELETING: `menuHapusData` (Shifting Array) 
void menuHapusData() {
    int cariNim, k = -1;
    cout << "\nMasukkan NIM yang ingin dihapus: ";
    if (!(cin >> cariNim)) {
        cout << ">> [ERROR] Input NIM harus angka!" << endl;
        clearBuffer(); return;
    }

    Mahasiswa daftar[MAX_MAHASISWA];
    int totalData = loadDataToArray(daftar);

    // Linear Search untuk mencari indeks elemen
    for (int i = 0; i < totalData; i++) {
        if (daftar[i].nim == cariNim) {
            k = i; break;
        }
    }

    if (k != -1) {
        // Menggeser sisa elemen ke kiri (Shifting Array)
        for (int i = k; i < totalData - 1; i++) {
            daftar[i] = daftar[i + 1];
        }
        totalData--; // Kurangi jumlah data asli
        tulisSemuaKeFile(daftar, totalData); // Tulis ulang file
        cout << ">> [SUCCESS] Mahasiswa dengan NIM " << cariNim << " berhasil dihapus." << endl;
    } else {
        cout << ">> [INFO] NIM " << cariNim << " tidak terdaftar di database." << endl;
    }
}
7. BUBBLE SORT & LOGIKA CARI DATA 
void bubbleSort(Mahasiswa arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j].nim > arr[j + 1].nim) {
                Mahasiswa temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void menuCariData() {
    int cariNim;
    cout << "\nMasukkan NIM yang dicari: ";
    if (!(cin >> cariNim)) {
        cout << ">> Input NIM harus angka." << endl; clearBuffer(); return;
    }
    Mahasiswa daftar[MAX_MAHASISWA];
    int totalData = loadDataToArray(daftar);
    bool ketemu = false;
    for (int i = 0; i < totalData; i++) {
        if (daftar[i].nim == cariNim) {
            cout << "\n>> Data Mahasiswa Ditemukan: \n--------------------------------------------------\n";
            cetakData(daftar[i]); cout << "--------------------------------------------------\n";
            ketemu = true; break;
        }
    }
    if (!ketemu) cout << ">> NIM " << cariNim << " tidak terdaftar." << endl;
}
8. KATEGORI SELEKSI: `menuTampilkanSemua` 
void menuTampilkanSemua() {
    Mahasiswa daftar[MAX_MAHASISWA];
    int totalData = loadDataToArray(daftar);
    if (totalData == 0) {
        cout << "\n>> Database kosong atau file tidak ditemukan." << endl; return;
    }
    bubbleSort(daftar, totalData);
    cout << "\n==================================================\n        KATEGORI: LOLOS SELEKSI BERKAS MAGANG\n==================================================\n";
    int countLolos = 0;
    for (int i = 0; i < totalData; i++) {
        if (daftar[i].semester >= 5 && daftar[i].ipk >= 3.25) {
            cetakData(daftar[i]); cout << "--------------------------------------------------" << endl;
            countLolos++;
        }
    }
    cout << "\n==================================================\n       KATEGORI: BELUM LOLOS SELEKSI BERKAS\n==================================================\n";
    for (int i = 0; i < totalData; i++) {
        if (daftar[i].semester < 5 || daftar[i].ipk < 3.25) {
            cetakData(daftar[i]); cout << "--------------------------------------------------" << endl;
        }
    }
}

void submenuOutput() {

⚠️ Penyelamat Buffer: Kalayan ngagunakeun separator spasi, prosés maca file ayeuna kantun nganggo operator >> sacara langsung tanpa hariwang kana buffer string anu kaluncatan.
💡 Konversi Spasi ka _: Fungsi for(char &c : s) if(c == ' ') c = '_' dianggo nalika input supados spasi dina nami henteu ngaruksak separator file, teras dibalikkeun deui janten spasi biasa nalika dicitak.
🚀 Logika Hapus Data (Shifting): Logika -> milari indeks target nganggo *Linear Search*, teras ngageser sésa data array ka kénca (arr[i] = arr[i+1]) sarta nyerat ulang file database tanpa mode append.
